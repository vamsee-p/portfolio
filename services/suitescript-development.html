/**
 * @NApiVersion 2.1
 * @NScriptType UserEventScript
 * Auto-assign approver based on PO amount
 */
define(['N/record', 'N/log'], (record, log) => {

  const beforeSubmit = (ctx) => {
    if (ctx.type !== ctx.UserEventType.CREATE) return;

    const amount = ctx.newRecord.getValue({
      fieldId: 'amount'
    });

    // Route to senior approver above $10k
    const approver = amount > 10000
      ? 'SENIOR_MGR_ID'
      : 'MGR_ID';

    ctx.newRecord.setValue({
      fieldId: 'custbody_approver',
      value: approver
    });
  };

  return { beforeSubmit };
});
